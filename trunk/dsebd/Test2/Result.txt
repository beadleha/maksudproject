	function getUrlValue(key){
		var value = "";
		for (var i=0;i<urlKeys.length;i++)
		{
			if (urlKeys[i]==key)
			{
				value = urlValues[i];
				break;
			}
		}
		return value;
	}

	function parseUrlString(){
		var url = window.location.search.substring(1);
		var kvpairs = url.split("&");
		for (var i = 0; i < kvpairs.length; i++)
		{
			var pos = kvpairs[i].indexOf('=');
			if (pos >= 0)
			{
				var argname = kvpairs[i].substring(0,pos);
				var value = kvpairs[i].substring(pos+1);
				urlKeys[urlKeys.length] = argname;
				urlValues[urlValues.length] = value;		
			}
		}
	}
